void main(){
    int<10>vector;

    for(int i=0;i<5; i = i+1){
        vector[i] = i; //0,1,2,3,4
    }

    for(int i=5;i<10;i=i+1){
        vector[i] = 5-i; //0,-1,-2,-3,-4
    }

    int max = vector[0];
    for(int i=0;i<10;i=i+1){
        if(vector[i]>max){
            max = vector[i];
        }
    }
    print(max);
}
